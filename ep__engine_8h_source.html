00912     tap_event_t walkTapQueue(const void *cookie, item **itm, void **es,
00913                              uint16_t *nes, uint8_t *ttl, uint16_t *flags,
00914                              uint32_t *seqno, uint16_t *vbucket) {
00915         LockHolder lh(tapNotifySync);
00916         std::map<const void*, TapConnection*>::iterator iter;
00917         TapConnection *connection = NULL;
00918         iter = tapConnectionMap.find(cookie);
00919         if (iter == tapConnectionMap.end()) {
00920             getLogger()->log(EXTENSION_LOG_WARNING, NULL,
00921                              "Walking a non-existent tap queue, disconnecting\n");
00922             return TAP_DISCONNECT;
00923         } else {
00924             connection = iter->second;
00925         }
00926 
00927         if (connection->doRunBackfill) {
00928             queueBackfill(connection, cookie);
00929         }
00930 
00931         tap_event_t ret = TAP_PAUSE;
00932         connection->paused = false;
00933 
00934         *es = NULL;
00935         *nes = 0;
00936         *ttl = (uint8_t)-1;
00937         *seqno = 0;
00938         *flags = 0;
00939 
00940         TapVBucketEvent ev = connection->nextVBucketHighPriority();
00941         if (ev.event != TAP_PAUSE) {
00942             *vbucket = ev.vbucket;
00943             *flags = static_cast<uint16_t>(ev.state);
00944             return ev.event;
00945         }
00946 
00947         if (!connection->empty()) {
00948             QueuedItem qi = connection->next();
00949             lh.unlock();
00950 
00951             ENGINE_ERROR_CODE r;
00952             *vbucket = qi.getVBucketId();
00953             std::string key = qi.getKey();
00954             r = get(cookie, itm, key.c_str(), (int)key.length(), qi.getVBucketId());
00955             if (r == ENGINE_SUCCESS) {
00956                 ret = TAP_MUTATION;
00957             } else if (r == ENGINE_KEY_ENOENT) {
00958                 ret = TAP_DELETION;
00959                 r = itemAllocate(cookie, itm,
00960                                  key.c_str(), key.length(), 0, 0, 0);
00961                 if (r != ENGINE_SUCCESS) {
00962                     EXTENSION_LOGGER_DESCRIPTOR *logger;
00963                     logger = (EXTENSION_LOGGER_DESCRIPTOR*)serverApi->extension->get_extension(EXTENSION_LOGGER);
00964 
00965                     getLogger()->log(EXTENSION_LOG_WARNING, NULL,
00966                                      "Failed to allocate memory for deletion of: %s\n", key.c_str());
00967                     ret = TAP_PAUSE;
00968                 }
00969             }
00970         } else if (connection->shouldFlush()) {
00971             ret = TAP_FLUSH;
00972         } else if (connection->shouldNoop()) {
00973             ret = TAP_NOOP;
00974         } else {
00975             connection->paused = true;
00976         }
00977 
00978         if (ret == TAP_PAUSE && connection->complete()) {
00979             ev = connection->nextVBucketLowPriority();

00981                 connection->paused = false;

00985                     epstore->setVBucketState(ev.vbucket, dead, serverApi->core);
00986                 }
00987                 return ev.event;
00988             }
00989 
00990             ret = TAP_DISCONNECT;
00991         }
00992 
00993         return ret;
00994     }

01360 
01361         if (ep_current_time() > nextTapNoop && tapIdleTimeout != (size_t)-1) {
01362             addNoop = true;
01363             nextTapNoop = ep_current_time() + (tapIdleTimeout / 2);

01370         std::map<const void*, TapConnection*>::iterator iter;
01371         for (iter = tapConnectionMap.begin(); iter != tapConnectionMap.end(); iter++) {
