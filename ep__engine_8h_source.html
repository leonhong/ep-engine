00808                     val += delta;
00809                 } else {

00811                         val = 0;
00812                     } else {
00813                         val -= delta;
00814                     }
00815                 }

00818                 vals << val << "\r\n";
00819                 size_t nb = vals.str().length();

00822                                      exptime, vals.str().c_str(), nb);

00825                 delete nit;

00828             }

00830             delete item;
00831         } else if (ret == ENGINE_NOT_MY_VBUCKET) {

01037                 tc->backfillAge = backfillAge;

01042                 tc->pendingBackfill = true;
01043             }

01045             if (flags & TAP_CONNECT_FLAG_LIST_VBUCKETS) {
01046                 tc->vbucketFilter = VBucketFilter(vbuckets);

01054                     TapVBucketEvent lo(TAP_VBUCKET_SET, *it, active);
01055                     tc->addVBucketHighPriority(hi);
01056                     tc->addVBucketLowPriority(lo);

01067                                 void *engine_specific,
01068                                 uint16_t nengine,
01069                                 uint8_t ttl,
01070                                 uint16_t tap_flags,
01071                                 tap_event_t tap_event,
01072                                 uint32_t tap_seqno,
01073                                 const void *key,
01074                                 size_t nkey,
01075                                 uint32_t flags,
01076                                 uint32_t exptime,

01078                                 const void *data,
01079                                 size_t ndata,
01080                                 uint16_t vbucket)

01082         (void)engine_specific;
01083         (void)nengine;
01084         (void)ttl;
01085         (void)tap_flags;
01086         (void)tap_seqno;

01089         // should disconnect it if tap isn't enabled.
01090         if (!cookie && !tapEnabled) {
01091             return ENGINE_DISCONNECT;

01093

01095         case TAP_FLUSH:

01105                 // We don't get the trailing CRLF in tap mutation but should store it

01114                 s->reserve(ndata+2);
01115                 s->append(static_cast<const char*>(data), ndata);
01116                 s->append("\r\n");
01117 
01118                 Item *item = new Item(k, flags, exptime, s);
01119                 item->setVBucketId(vbucket);
01120 
01121                 /* @TODO we don't have CAS now.. we might in the future.. */
01122                 (void)cas;
01123                 ENGINE_ERROR_CODE ret = epstore->set(*item, cookie, true);
01124                 if (ret == ENGINE_SUCCESS) {
01125                     addMutationEvent(item, vbucket);
01126                 }
01127 
01128                 delete item;
01129                 return ret;
01130             }
01131 
01132         case TAP_OPAQUE:
01133             break;
01134 
01135         case TAP_VBUCKET_SET:
01136             {
01137                 vbucket_state_t state = static_cast<vbucket_state_t>(tap_flags);
01138                 epstore->setVBucketState(vbucket, state, serverApi->core);
01139                 if (state == active) {
01140                     return ENGINE_DISCONNECT;
01141                 }
01142             }
01143             break;
01144 
01145         default:
01146             abort();
01147         }
01148 
01149         return ENGINE_SUCCESS;
01150     }

01302     void notifyTapIoThreadMain(void) {
01303         bool addNoop = false;
01304 
01305         if (ep_current_time() > nextTapNoop) {
01306             addNoop = true;
01307             nextTapNoop = ep_current_time() + (tapIdleTimeout / 2);
01308         }
01309         LockHolder lh(tapNotifySync);
01310         // We should pause unless we purged some connections or
01311         // all queues have items.
01312         bool shouldPause = purgeExpiredTapConnections_UNLOCKED() == 0;
01313         // see if I have some channels that I have to signal..
01314         std::map<const void*, TapConnection*>::iterator iter;
01315         for (iter = tapConnectionMap.begin(); iter != tapConnectionMap.end(); iter++) {
01316             if (!iter->second->queue->empty()) {
01317                 shouldPause = false;
01318             } else if (addNoop) {

01320                 iter->second->addVBucketHighPriority(hi);

01340         std::map<const void*, TapConnection*> tcm = tapConnectionMap;
